// see common.txt for shared structures such as string

VERSION 3.0

owmat:
    uint16 version_major = 3;
    uint16 version_minor = 1;
    uint32 enum { material = 0, modellook = 1 } type;

    if (type == material) {
        material material;
    } else if (type == modellook) {
        modellook modellook;
    }

material:
    uint64 texture_count;
    uint64 static_count;
    uint32 shader_id;
    texture textures[texture_count];
    static statics[static_count];

texture:
    string texture_path;
    uint32 texture_id;

static:
    uint32 static_id;
    uint32 data_size;
    uint8 data[data_size];

modellook:
    uint64 count;
    modellook_entry[count] entries;
    uint64 geoset_count;
    uint32[geoset_count] geoset_entries;

modellook_entry:
    uint64 id;
    string material_path;
